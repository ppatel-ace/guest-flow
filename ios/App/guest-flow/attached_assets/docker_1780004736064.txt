version: '3.8'

services:
  app:
    image: guestflow-guestflow:latest
    build: .
    container_name: guestflow-app
    restart: always
    # CHANGED: Swapped 'expose' to 'ports' so Docker maps it securely inside the bridge
    ports:
      - "5000:5000"
    environment:
      NODE_ENV: production
      PORT: 5000
      DATABASE_URL: ${DATABASE_URL}
      SESSION_SECRET: ${SESSION_SECRET:-change_this_to_a_long_random_secret}
      FINGERPRINT_HMAC_SECRET: ${FINGERPRINT_HMAC_SECRET:-change_this_to_a_long_random_secret}
      TURNSTILE_SECRET_KEY: ${TURNSTILE_SECRET_KEY:-}
      VITE_TURNSTILE_SITE_KEY: ${VITE_TURNSTILE_SITE_KEY:-}
    networks:
      guestflow_net:
        ipv4_address: 10.20.11.2

  nginx:
    image: nginx:alpine
    container_name: guestflow-nginx
    restart: always
    ports:
      - "6055:80" # Maps physical server port 6055 to Nginx port 80
    volumes:
      - ./nginx.conf:/etc/nginx/conf.d/default.conf:ro
    depends_on:
      - app
    networks:
      guestflow_net:
        ipv4_address: 10.20.11.3

networks:
  guestflow_net:
    driver: bridge
    # ADDED: This forces Linux to keep the bridge up and functional
    driver_opts:
      com.docker.network.bridge.name: br-guestflow
    ipam:
      config:
        - subnet: 10.20.11.0/24
          gateway: 10.20.11.1
